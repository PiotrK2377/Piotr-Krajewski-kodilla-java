SELECT L.NAME, I.SUMMARY FROM ISSUELISTS L
JOIN ISSUES I ON L.ID = I.ISSUESLIST_ID;

SELECT U.FIRSTNAME, U.LASTNAME, I.SUMMARY FROM USERS U
JOIN ISSUES I on U.ID = I.USER_ID_ASSIGNEDTO;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS ISSUES_NUMBER
FROM USERS U JOIN ISSUES I on U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY I.USER_ID_ASSIGNEDTO
HAVING COUNT(*) > 0;
